<body>

    <div id="player-container"></div>


    <script>

        // Deine YouTube Video-IDs

        const videoIds = [

            "3X_BYb9ws9Q", // Zábavní bazén

            "VmGrd3VlA2Y", // Aquapark UH

            "pPJKse_k50s"  // Armdale Roundabout

        ];


        let currentIndex = 0;

        const displayTime = 120000; // 120000 ms = 2 Minuten


        function switchStream() {

            const container = document.getElementById('player-container');

            const id = videoIds[currentIndex];

            

            // YouTube Embed URL mit Parametern für Autoplay, Mute und ohne UI

            const embedUrl = `https://www.youtube.com/embed/${id}?autoplay=1&mute=1&controls=0&showinfo=0&rel=0&iv_load_policy=3&modestbranding=1&disablekb=1&playlist=${id}&loop=1`;

            

            container.innerHTML = `<iframe src="${embedUrl}" allow="autoplay; encrypted-media; fullscreen"></iframe>`;


            // Index für das nächste Video erhöhen

            currentIndex = (currentIndex + 1) % videoIds.length;

        }


        // Intervall für den automatischen Wechsel starten

        setInterval(switchStream, displayTime);

        

        // Den ersten Stream sofort laden

        switchStream();

    </script>

</body>
